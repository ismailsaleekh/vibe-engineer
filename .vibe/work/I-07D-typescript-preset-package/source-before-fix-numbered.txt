   548	export function renderTypeScriptPresetFiles(options: RenderTypeScriptPresetOptions = {}): readonly GeneratedPresetFile[] {
   549	  const packageDirectory = normalizePackageDirectory(options.packageDirectory);
   550	  const packageName = normalizePackageName(options.packageName);
   551	  const includeSampleSource = options.includeSampleSource ?? true;
   552	  const files = [
   553	    renderedFile("tsconfig.base.json", "typescript-config", rootTsconfigContent()),
   554	    renderedFile(`${packageDirectory}/tsconfig.json`, "typescript-config", packageTsconfigContent()),
   555	    renderedFile("eslint.config.mjs", "eslint-config", renderEslintConfig()),
   556	    renderedFile(".vibe/generated/typescript-preset/eslint-policy.json", "eslint-policy", stableStringify(ESLINT_POLICY_DEFAULTS)),
   557	    renderedFile("prettier.config.mjs", "prettier-config", renderPrettierConfig()),
   558	    renderedFile(".vibe/generated/typescript-preset/prettier-policy.json", "prettier-policy", stableStringify(PRETTIER_DEFAULTS)),
   559	    renderedFile("pnpm-workspace.yaml", "workspace-config", pnpmWorkspaceContent()),
   560	    renderedFile("turbo.json", "turbo-config", turboContent()),
   561	    renderedFile("package.json", "package-manifest", packageJsonContent()),

   564	  ];

   568	  return Object.freeze([...files, renderedFile(`${packageDirectory}/src/sample.ts`, "source-fixture", sampleSourceContent())]);

   583	  return value !== null && typeof value === "object" && !Array.isArray(value);
   584	}

   590	function parseJsonObject(content: string, path: string, findings: TypeScriptPresetFinding[]): UnknownObject | undefined {
   591	  let parsed: unknown;
   592	  try {
   593	    parsed = JSON.parse(content);
   594	  } catch (error) {
   595	    findings.push(makeFinding("PRESET_MALFORMED_JSON", path, "Generated JSON content is malformed.", [error instanceof Error ? error.message : "unknown parse error"]));
   596	    return undefined;
   597	  }
   598	  if (!isObject(parsed)) {
   599	    findings.push(makeFinding("PRESET_MALFORMED_JSON", path, "Generated JSON content must be an object.", []));
   600	    return undefined;
   601	  }
   602	  return parsed;
   603	}

   606	  code: TypeScriptPresetFindingCode,
   607	  path: string,
   608	  message: string,
   609	  evidence: readonly string[],
   610	): TypeScriptPresetFinding {

   614	function isSafeRelativePath(path: string): boolean {
   615	  if (path.length === 0 || path.startsWith("/") || path.startsWith("\\") || path.includes("\\")) {
   616	    return false;
   617	  }
   618	  const segments = path.split("/");
   619	  return segments.every((segment) => segment.length > 0 && segment !== "." && segment !== "..");
   620	}

   622	function validatePath(path: string, seen: Set<string>, findings: TypeScriptPresetFinding[]): void {
   623	  if (!isSafeRelativePath(path)) {
   624	    findings.push(makeFinding("PRESET_UNSAFE_GENERATED_PATH", path, "Generated file path must be normalized, relative, and contained inside the target root.", [path]));
   625	  }
   626	  if (seen.has(path)) {
   627	    findings.push(makeFinding("PRESET_DUPLICATE_GENERATED_PATH", path, "Generated file path appears more than once.", [path]));
   628	  }
   629	  seen.add(path);
   630	}

   632	function validateStrictTsconfig(content: string | undefined, path: string, findings: TypeScriptPresetFinding[]): void {
   633	  if (content === undefined) {
   634	    findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", path, "Required strict tsconfig file is missing.", []));
   635	    return;
   636	  }
   637	  const root = parseJsonObject(content, path, findings);
   638	  if (root === undefined) return;
   639	  const compilerOptions = getObjectProperty(root, "compilerOptions");
   640	  if (!isObject(compilerOptions)) {
   641	    findings.push(makeFinding("PRESET_STRICT_FLAG_WEAKENED", path, "tsconfig compilerOptions object is missing.", []));
   642	    return;
   643	  }
   644	  for (const flag of REQUIRED_STRICT_TRUE_FLAGS) {
   645	    if (getObjectProperty(compilerOptions, flag) !== true) {
   646	      findings.push(makeFinding("PRESET_STRICT_FLAG_WEAKENED", path, `Required TypeScript flag must be true: ${flag}.`, [flag]));
   647	    }
   648	  }
   649	  for (const flag of REQUIRED_STRICT_FALSE_FLAGS) {
   650	    if (getObjectProperty(compilerOptions, flag) !== false) {
   651	      findings.push(makeFinding("PRESET_STRICT_FLAG_WEAKENED", path, `Required TypeScript flag must be false: ${flag}.`, [flag]));
   652	    }
   653	  }
   654	  if (getObjectProperty(compilerOptions, "module") !== "NodeNext" || getObjectProperty(compilerOptions, "moduleResolution") !== "NodeNext") {
   655	    findings.push(makeFinding("PRESET_STRICT_FLAG_WEAKENED", path, "TypeScript module and moduleResolution must both be NodeNext.", []));
   656	  }
   657	}

   659	function validatePackageTsconfig(content: string | undefined, path: string, findings: TypeScriptPresetFinding[]): void {
   660	  if (content === undefined) {
   661	    findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", path, "Required package tsconfig is missing.", []));
   662	    return;
   663	  }
   664	  const root = parseJsonObject(content, path, findings);
   665	  if (root === undefined) return;
   666	  if (getObjectProperty(root, "extends") !== "../../tsconfig.base.json") {
   667	    findings.push(makeFinding("PRESET_PACKAGE_TSCONFIG_WEAKENED", path, "Package tsconfig must extend ../../tsconfig.base.json.", []));
   668	  }
   669	  const compilerOptions = getObjectProperty(root, "compilerOptions");
   670	  if (isObject(compilerOptions)) {
   671	    for (const flag of [...REQUIRED_STRICT_TRUE_FLAGS, ...REQUIRED_STRICT_FALSE_FLAGS]) {
   672	      if (Object.hasOwn(compilerOptions, flag)) {
   673	        findings.push(makeFinding("PRESET_PACKAGE_TSCONFIG_WEAKENED", path, `Package tsconfig must not override strict base flag: ${flag}.`, [flag]));
   674	      }
   675	    }
   676	  }
   677	}

   679	function validateEslintPolicy(content: string | undefined, path: string, findings: TypeScriptPresetFinding[]): void {
   680	  if (content === undefined) {
   681	    findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", path, "Required ESLint policy file is missing.", []));
   682	    return;
   683	  }
   684	  const policy = parseJsonObject(content, path, findings);
   685	  if (policy === undefined) return;
   686	  const rules = getObjectProperty(policy, "rules");
   687	  if (!Array.isArray(rules)) {
   688	    findings.push(makeFinding("PRESET_ESLINT_RULE_MISSING", path, "ESLint policy rules array is missing.", []));
   689	    return;
   690	  }
   691	  const ruleNames = new Set<string>();
   692	  for (const rule of rules) {
   693	    if (isObject(rule)) {
   694	      const name = getObjectProperty(rule, "name");
   695	      const severity = getObjectProperty(rule, "severity");
   696	      if (typeof name === "string" && severity === "error") {
   697	        ruleNames.add(name);
   698	      }
   699	    }
   700	  }
   701	  for (const requiredRule of REQUIRED_ESLINT_RULES) {
   702	    if (!ruleNames.has(requiredRule)) {
   703	      findings.push(makeFinding("PRESET_ESLINT_RULE_MISSING", path, `Required ESLint policy rule is missing: ${requiredRule}.`, [requiredRule]));
   704	    }
   705	  }
   706	  const boundaryRules = getObjectProperty(policy, "boundaryRules");
   707	  if (!isObject(boundaryRules)
   708	    || getObjectProperty(boundaryRules, "rawJsonParseRequiresNamedBoundary") !== true
   709	    || getObjectProperty(boundaryRules, "broadDomainMapModelsForbidden") !== true) {
   710	    findings.push(makeFinding("PRESET_ESLINT_RULE_MISSING", path, "ESLint boundary rules for JSON.parse and broad domain map models are required.", []));
   711	  }
   712	}

   744	function validateDomainNeutralText(files: readonly ValidatorFile[], findings: TypeScriptPresetFinding[]): void {

   756	function validateManifestContent(content: string | undefined, path: string, findings: TypeScriptPresetFinding[]): void {
   757	  if (content === undefined) {
   758	    findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", path, "Required generated preset manifest is missing.", []));
   759	    return;
   760	  }
   761	  const manifest = parseJsonObject(content, path, findings);
   762	  if (manifest === undefined) return;
   763	  const metadata = getObjectProperty(manifest, "metadata");
   764	  const files = getObjectProperty(manifest, "files");
   765	  if (!isObject(metadata) || getObjectProperty(metadata, "presetId") !== TYPE_SCRIPT_PRESET_ID || !Array.isArray(files)) {
   766	    findings.push(makeFinding("PRESET_MANIFEST_CONTENT_MISMATCH", path, "Generated manifest metadata/files contract does not match the preset API.", []));
   767	    return;
   768	  }
   769	  const manifestPaths = new Set<string>();
   770	  for (const entry of files) {
   771	    if (isObject(entry)) {
   772	      const outputPath = getObjectProperty(entry, "outputPath");
   773	      if (typeof outputPath === "string") {
   774	        manifestPaths.add(outputPath);
   775	      }
   776	    }
   777	  }
   778	  for (const required of TYPESCRIPT_PRESET_FILE_MANIFEST) {
   779	    if (!manifestPaths.has(required.outputPath)) {
   780	      findings.push(makeFinding("PRESET_MANIFEST_CONTENT_MISMATCH", path, `Generated manifest omits required output path: ${required.outputPath}.`, [required.outputPath]));
   781	    }
   782	  }
   783	}

   785	function validatePackageManifest(content: string | undefined, path: string, findings: TypeScriptPresetFinding[]): void {
   786	  if (content === undefined) {
   787	    findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", path, "Required package manifest is missing.", []));
   788	    return;
   789	  }
   790	  const manifest = parseJsonObject(content, path, findings);
   791	  if (manifest === undefined) return;
   792	  for (const dependencyField of ["dependencies", "devDependencies", "peerDependencies", "optionalDependencies"]) {
   793	    const dependencies = getObjectProperty(manifest, dependencyField);
   794	    if (!isObject(dependencies)) continue;
   795	    if (getObjectProperty(dependencies, "@vibe-engineer/testing") !== undefined && dependencyField === "dependencies") {
   796	      findings.push(makeFinding("PRESET_FORBIDDEN_TESTING_PRODUCTION_DEPENDENCY", path, "Generated production dependencies must not include @vibe-engineer/testing.", [dependencyField]));
   797	    }
   798	  }

   801	export function validateTypeScriptPresetFiles(filesInput: unknown): TypeScriptPresetValidationResult {
   802	  const findings: TypeScriptPresetFinding[] = [];
   803	  if (!Array.isArray(filesInput)) {
   804	    findings.push(makeFinding("PRESET_MALFORMED_RENDER_OPTIONS", "", "Validator input must be an array of generated preset files.", []));
   805	    return { ok: false, findings: Object.freeze(findings), fileCount: 0 };
   806	  }
   807	
   808	  const files: ValidatorFile[] = [];
   809	  for (const item of filesInput) {
   810	    if (!isObject(item)) {
   811	      findings.push(makeFinding("PRESET_MALFORMED_RENDER_OPTIONS", "", "Generated preset file entry must be an object.", []));
   812	      continue;
   813	    }
   814	    const path = getObjectProperty(item, "path");
   815	    const kind = getObjectProperty(item, "kind");
   816	    const content = getObjectProperty(item, "content");
   817	    const manifest = getObjectProperty(item, "manifest");
   818	    if (typeof path !== "string" || typeof kind !== "string" || typeof content !== "string" || !isObject(manifest)) {
   819	      findings.push(makeFinding("PRESET_MALFORMED_RENDER_OPTIONS", "", "Generated preset file entry has malformed path, kind, content, or manifest.", []));
   820	      continue;
   821	    }
   822	    const manifestOutputPath = getObjectProperty(manifest, "outputPath");
   823	    if (manifestOutputPath !== path) {
   824	      findings.push(makeFinding("PRESET_MANIFEST_CONTENT_MISMATCH", path, "Generated file manifest outputPath must match file path.", []));
   825	    }
   826	    if (!TYPESCRIPT_PRESET_FILE_MANIFEST.some((entry) => entry.kind === kind)) {
   827	      findings.push(makeFinding("PRESET_MALFORMED_RENDER_OPTIONS", path, "Generated preset file entry has an unknown kind.", [kind]));
   828	    }
   829	    files.push({ path, content });
   830	  }
   831	
   832	  const seen = new Set<string>();
   833	  const byPath = new Map<string, string>();
   834	  for (const file of files) {
   835	    validatePath(file.path, seen, findings);
   836	    byPath.set(file.path, file.content);
   837	  }
   838	
   839	  for (const requiredPath of REQUIRED_FILE_PATHS) {
   840	    if (!byPath.has(requiredPath)) {
   841	      findings.push(makeFinding("PRESET_MISSING_REQUIRED_FILE", requiredPath, "Required generated file is missing.", [requiredPath]));
   842	    }
   843	  }
   844	
   845	  validateStrictTsconfig(byPath.get("tsconfig.base.json"), "tsconfig.base.json", findings);
   846	  validatePackageTsconfig(byPath.get("packages/example/tsconfig.json"), "packages/example/tsconfig.json", findings);
   847	  validateEslintPolicy(byPath.get(".vibe/generated/typescript-preset/eslint-policy.json"), ".vibe/generated/typescript-preset/eslint-policy.json", findings);
   848	  validatePrettierPolicy(byPath.get(".vibe/generated/typescript-preset/prettier-policy.json"), ".vibe/generated/typescript-preset/prettier-policy.json", findings);
   849	  validateQuickGate(byPath.get("package.json"), "package.json", findings);
   850	  validateQuickGate(byPath.get("turbo.json"), "turbo.json", findings);
   851	  validatePackageManifest(byPath.get("package.json"), "package.json", findings);
   852	  validatePackageManifest(byPath.get("packages/example/package.json"), "packages/example/package.json", findings);
   853	  validateManifestContent(byPath.get(".vibe/generated/typescript-preset/manifest.json"), ".vibe/generated/typescript-preset/manifest.json", findings);
   854	  validateDomainNeutralText(files, findings);
   855	
   856	  if (findings.length > 0) {
   857	    return { ok: false, findings: Object.freeze(findings), fileCount: files.length };
   858	  }
   859	  return { ok: true, findings: Object.freeze([]), fileCount: files.length };
   860	}
